Sub HighlightSheetReferences()
    Dim ws As Worksheet
    Dim cell As Range
    Dim searchSheets As Variant
    Dim i As Integer
    Dim found As Boolean
    
    ' List of sheets to search for references
    searchSheets = Array("Sheet1!", "Sheet2!", "Sheet3!")
    
    For Each ws In ThisWorkbook.Worksheets
        For Each cell In ws.UsedRange
            If Not IsError(cell.Formula) Then
                For i = LBound(searchSheets) To UBound(searchSheets)
                    If InStr(1, cell.Formula, searchSheets(i), vbTextCompare) > 0 Then
                        found = True
                        cell.Interior.Color = RGB(255, 0, 0) ' Highlight in red
                        Exit For ' Exit the loop once a match is found
                    End If
                Next i
            End If
        Next cell
    Next ws
    
    If found Then
        MsgBox "Cells referencing the specified sheets have been highlighted in red."
    Else
        MsgBox "No references to the specified sheets found."
    End If
End Sub

Sub HighlightSheetReferences()
    Dim ws As Worksheet
    Dim cell As Range
    Dim searchSheets As Variant
    Dim i As Integer
    Dim found As Boolean
    
    ' List of sheets to search for references
    searchSheets = Array("Sheet1!", "Sheet2!", "Sheet3!")
    
    On Error GoTo ErrorHandler
    
    For Each ws In ThisWorkbook.Worksheets
        ' Check if the sheet has any used range
        If Not ws.UsedRange Is Nothing Then
            For Each cell In ws.UsedRange
                If Not IsError(cell.Formula) Then
                    For i = LBound(searchSheets) To UBound(searchSheets)
                        If InStr(1, cell.Formula, searchSheets(i), vbTextCompare) > 0 Then
                            found = True
                            cell.Interior.Color = RGB(255, 0, 0) ' Highlight in red
                            Exit For ' Exit the loop once a match is found
                        End If
                    Next i
                End If
            Next cell
        End If
    Next ws
    
    If found Then
        MsgBox "Cells referencing the specified sheets have been highlighted in red."
    Else
        MsgBox "No references to the specified sheets found."
    End If
    
    Exit Sub
    
ErrorHandler:
    MsgBox "An error occurred: " & Err.Description
End Sub
