# Create Excel COM object
$excel = New-Object -ComObject Excel.Application

# Optional: Make Excel visible (useful for debugging)
$excel.Visible = $true

# Define path to the Excel file
$filePath = "C:\Path\To\Your\File.xlsx"

# Open the workbook
$workbook = $excel.Workbooks.Open($filePath)

# Optional: Do something with the workbook, like modifying a cell
# $sheet = $workbook.Sheets.Item(1)
# $sheet.Cells.Item(1, 1).Value2 = "Edited via PowerShell"

# Save the workbook
$workbook.Save()

# Close the workbook
$workbook.Close($false)  # $false = don't prompt to save again

# Quit Excel
$excel.Quit()

# Release the COM objects
[System.Runtime.Interopservices.Marshal]::ReleaseComObject($workbook) | Out-Null
[System.Runtime.Interopservices.Marshal]::ReleaseComObject($excel) | Out-Null
[GC]::Collect()
[GC]::WaitForPendingFinalizers()
